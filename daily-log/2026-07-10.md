# 상속과 다형성을 활용한 배달 시스템 설계

## 모범 답안

### 문제: 조건문 지옥 (Before)
```python
def process_delivery(delivery_type, order):
    if delivery_type == "restaurant":
        delivery_time = 30
        fee = 3000
        # 음식점 배달 로직
    elif delivery_type == "convenience":
        delivery_time = 15
        fee = 2000
        # 편의점 배달 로직
    elif delivery_type == "pharmacy":
        delivery_time = 0  # 즉시
        fee = 5000
        # 약국 배달 로직
    # 새 타입 추가될 때마다 else-if 추가...
```

**문제점:**
- ❌ 새 배달 타입 추가 시 기존 함수 수정 필요 (OCP 위반)
- ❌ 로직이 한 곳에 집중되어 복잡도 증가
- ❌ 테스트하기 어려움
- ❌ 실수 가능성 높음

---

### 해결책: 상속과 다형성 (After)

```python
from abc import ABC, abstractmethod
from typing import List

# 1️⃣ 추상 기본 클래스 (부모)
class Delivery(ABC):
    """모든 배달 타입이 준수할 계약"""
    
    @abstractmethod
    def get_delivery_time(self) -> int:
        """배달 소요 시간 반환"""
        pass
    
    @abstractmethod
    def calculate_fee(self, distance: float) -> int:
        """배송비 계산"""
        pass
    
    @abstractmethod
    def validate_order(self, order) -> bool:
        """주문 유효성 검증"""
        pass
    
    def process(self, order):
        """공통 배달 처리 로직"""
        if not self.validate_order(order):
            raise ValueError("Invalid order")
        
        delivery_time = self.get_delivery_time()
        fee = self.calculate_fee(order.distance)
        
        return {
            "status": "delivery_confirmed",
            "estimated_time": delivery_time,
            "fee": fee
        }

# 2️⃣ 구체적인 구현 클래스들 (자식)
class RestaurantDelivery(Delivery):
    def get_delivery_time(self) -> int:
        return 30
    
    def calculate_fee(self, distance: float) -> int:
        base_fee = 3000
        additional_fee = int(distance * 500)  # 거리별 추가 요금
        return base_fee + additional_fee
    
    def validate_order(self, order) -> bool:
        return order.min_order_amount >= 10000

class ConvenienceDelivery(Delivery):
    def get_delivery_time(self) -> int:
        return 15
    
    def calculate_fee(self, distance: float) -> int:
        return 2000  # 고정요금
    
    def validate_order(self, order) -> bool:
        return order.min_order_amount >= 5000

class PharmacyDelivery(Delivery):
    def get_delivery_time(self) -> int:
        return 0  # 즉시
    
    def calculate_fee(self, distance: float) -> int:
        return 5000
    
    def validate_order(self, order) -> bool:
        # 의약품은 처방전 확인 필요
        return hasattr(order, 'prescription_id')

# 3️⃣ 배달 시스템 (클라이언트)
class DeliveryService:
    def __init__(self):
        self.deliveries: dict[str, type[Delivery]] = {
            "restaurant": RestaurantDelivery,
            "convenience": ConvenienceDelivery,
            "pharmacy": PharmacyDelivery,
        }
    
    def process_order(self, delivery_type: str, order) -> dict:
        """다형성 활용: 실제 타입을 몰라도 Delivery 인터페이스로 처리"""
        
        if delivery_type not in self.deliveries:
            raise ValueError(f"Unsupported delivery type: {delivery_type}")
        
        # 다형성: 어떤 Delivery 타입이든 같은 방식으로 처리
        delivery = self.deliveries[delivery_type]()
        return delivery.process(order)
    
    # 🎯 새로운 배달 타입 추가는 register_delivery로 간단히 해결
    def register_delivery(self, type_name: str, delivery_class: type[Delivery]):
        """런타임에 새 배달 타입 등록"""
        self.deliveries[type_name] = delivery_class

# 4️⃣ 사용 예시
class Order:
    def __init__(self, distance, min_order_amount):
        self.distance = distance
        self.min_order_amount = min_order_amount

# 테스트
service = DeliveryService()

restaurant_order = Order(distance=3.5, min_order_amount=15000)
result = service.process_order("restaurant", restaurant_order)
print(result)  # {'status': 'delivery_confirmed', 'estimated_time': 30, 'fee': 4750}

# 🎉 새로운 타입 추가: 기존 코드 수정 없음!
class LiquorDelivery(Delivery):
    def get_delivery_time(self) -> int:
        return 20
    
    def calculate_fee(self, distance: float) -> int:
        return 4000
    
    def validate_order(self, order) -> bool:
        return hasattr(order, 'age_verified') and order.age_verified

service.register_delivery("liquor", LiquorDelivery)
```

## 핵심 개념 정리

### 📌 **상속 (Inheritance)**
- **정의**: 부모 클래스의 속성과 메서드를 자식 클래스가 물려받음
- **목적**: 
  - 공통 코드 중복 제거 (DRY 원칙)
  - 계층 구조 표현
  
```python
# 부모가 정의한 계약(interface)을 모든 자식이 구현
class RestaurantDelivery(Delivery):  # Delivery 상속
    pass
```

### 🎭 **다형성 (Polymorphism)**
- **정의**: "하나의 인터페이스, 여러 구현"
  - 같은 메서드 이름이지만 클래스별로 다른 동작
  
```python
# 같은 process() 메서드를 호출하지만
# RestaurantDelivery, PharmacyDelivery마다 다르게 동작
delivery.process(order)  # 실제 타입에 따라 다른 결과
```

### 🏗️ **추상 클래스 (Abstract Base Class)**
```python
from abc import ABC, abstractmethod

class Delivery(ABC):
    @abstractmethod  # 자식 클래스가 반드시 구현해야 함
    def validate_order(self, order) -> bool:
        pass
```
- **이점**: 자식 클래스가 특정 메서드를 반드시 구현하도록 강제
- 계약 기반 프로그래밍 (Design by Contract)

### 🔑 **SOLID 원칙과의 연관**

| 원칙 | 설명 | 예시 |
|------|------|------|
| **O** (Open-Closed) | 확장에 열려있고, 수정에 닫혀있음 | 새 배달타입 추가 시 기존 코드 수정 없음 ✅ |
| **L** (Liskov Substitution) | 자식 클래스는 부모 자리에 대체 가능 | 어떤 Delivery 타입이든 process() 호출 가능 |
| **D** (Dependency Inversion) | 구체적 클래스 대신 추상화에 의존 | `Delivery` 추상 클래스에 의존 |
