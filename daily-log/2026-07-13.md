# 상속과 다형성을 활용한 배달앱 설계

## 모범 답안

이 문제는 **상속으로 공통 로직을 추상화**하고, **다형성으로 상태별 동작을 유연하게 처리**하는 전형적인 사례입니다.

### 1단계: 공통 추상화 (상속의 핵심)

```python
from abc import ABC, abstractmethod
from enum import Enum
from datetime import datetime

# 추상 기본 클래스: 모든 배송 주체의 공통 인터페이스 정의
class DeliveryEntity(ABC):
    def __init__(self, entity_id: str):
        self.entity_id = entity_id
        self.state = None
        self.state_history = []
    
    # 템플릿 메서드: 상태 변화의 일반적인 프로세스 정의
    def change_state(self, new_state):
        """상태 변경의 공통 로직 (중복 제거)"""
        if self._is_valid_transition(self.state, new_state):
            old_state = self.state
            self.state = new_state
            self.state_history.append({
                'from': old_state,
                'to': new_state,
                'timestamp': datetime.now()
            })
            
            # 다형성: 각 주체별 맞춤 동작
            self._on_state_changed(old_state, new_state)
            self._send_notification()  # 공통: 항상 실행
            self._update_related_data()  # 공통: 항상 실행
        else:
            raise InvalidStateTransition(f"{old_state} → {new_state}")
    
    # 추상 메서드: 구현체에서 반드시 정의해야 함
    @abstractmethod
    def _is_valid_transition(self, current: str, next: str) -> bool:
        """각 주체별 유효한 상태 전이만 허용"""
        pass
    
    @abstractmethod
    def _on_state_changed(self, old_state: str, new_state: str):
        """상태 변경 시 주체별 추가 로직"""
        pass
    
    @abstractmethod
    def get_notification_message(self) -> str:
        """주체별 맞춤 알림 메시지"""
        pass
    
    # 공통 메서드: 모든 구현체에서 동일
    def _send_notification(self):
        """공통 알림 로직"""
        message = self.get_notification_message()
        NotificationService.send(self.entity_id, message)
    
    def _update_related_data(self):
        """상태 변경 시 관련 데이터 업데이트"""
        OrderStatusCache.update(self.entity_id, self.state)
```

class Restaurant(DeliveryEntity):
    def __init__(self, restaurant_id: str):
        super().__init__(restaurant_id)
        self.state = "preparing"  # 초기 상태
    
    def _is_valid_transition(self, current: str, next: str) -> bool:
        """음식점의 유효한 상태 전이"""
        valid_transitions = {
            "preparing": ["completed"],
            "completed": ["pickup_completed"],
            "pickup_completed": []
        }
        return next in valid_transitions.get(current, [])
    
    def _on_state_changed(self, old_state: str, new_state: str):
        """음식점 특화 로직"""
        if new_state == "completed":
            # 라이더에게 픽업 준비 완료 알림
            self._notify_available_riders()
        elif new_state == "pickup_completed":
            # 주방 통계 업데이트
