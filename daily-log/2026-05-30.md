# ✅ 리팩토링 후: SRP + OCP 적용

# 1단계: 책임 분리 - 각 기능을 독립적인 인터페이스로
from abc import ABC, abstractmethod

# 결제 책임
class PaymentProcessor(ABC):
    @abstractmethod
    def pay(self, order: Order) -> PaymentResult:
        pass

class CreditCardPaymentProcessor(PaymentProcessor):
    def pay(self, order: Order) -> PaymentResult:
        # 신용카드 결제 로직
        return PaymentResult(success=True)

class KakaoPayProcessor(PaymentProcessor):
    def pay(self, order: Order) -> PaymentResult:
        # 카카오페이 결제 로직
        return PaymentResult(success=True)

class NaverPayProcessor(PaymentProcessor):
    def pay(self, order: Order) -> PaymentResult:
        # 네이버페이 결제 로직 (새로운 기능 추가!)
        return PaymentResult(success=True)


# 알림 책임
class NotificationService(ABC):
    @abstractmethod
    def notify(self, customer: Customer, message: str) -> bool:
        pass

class EmailNotificationService(NotificationService):
    def notify(self, customer: Customer, message: str) -> bool:
        # 이메일 발송
        return True

class SMSNotificationService(NotificationService):
    def notify(self, customer: Customer, message: str) -> bool:
        # SMS 발송
        return True

class KakaoTalkNotificationService(NotificationService):
    def notify(self, customer: Customer, message: str) -> bool:
        # 카카오톡 발송 (새로운 기능 추가!)
        return True


# 배달 추적 책임
class DeliveryTracker(ABC):
    @abstractmethod
    def track(self, order: Order) -> None:
        pass

class BasicDeliveryTracker(DeliveryTracker):
    def track(self, order: Order) -> None:
        # 기본 배달 추적
        pass

class RealtimeGPSTracker(DeliveryTracker):
    def track(self, order: Order) -> None:
        # 실시간 GPS 추적 (새로운 방식 추가!)
        pass


# 주문 저장 책임
class OrderRepository(ABC):
    @abstractmethod
    def save(self, order: Order) -> None:
        pass

class DatabaseOrderRepository(OrderRepository):
    def save(self, order: Order) -> None:
        # DB 저장
        pass


# 2단계: 핵심 비즈니스 로직은 간단하게 - Orchestrator 패턴
class OrderService:
    """
    SRP: 오직 주문 처리의 흐름 조정만 담당
    의존성은 모두 주입받음 (DI)
    """
    def __init__(
        self,
        payment_processor: PaymentProcessor,
        notifiers: List[NotificationService],
        delivery_tracker: DeliveryTracker,
        order_repository: OrderRepository
    ):
        self.payment_processor = payment_processor
        self.notifiers = notifiers
        self.delivery_tracker = delivery_tracker
        self.order_repository = order_repository

    def process_order(self, order: Order) -> bool:
        try:
            # 1. 주문 저장
            self.order_repository.save(order)
            
            # 2. 결제 처리 (구체적인 결제 방식을 모름)
            payment_result = self.payment_processor.pay(order)
            if not payment_result.success:
                self._notify_all(f"결제 실패: {payment_result.error}")
                return False
            
            # 3. 배달 시작
            self.delivery_tracker.track(order)
            
            # 4. 고객 알림 (모든 채널에 동시에 발송)
            self._notify_all(f"주문 {order.id} 접수되었습니다")
            
            return True
        except Exception as e:
            self._notify_all(f"주문 처리 중 오류 발생: {str(e)}")
            return False

    def _notify_all(self, message: str) -> None:
        """모든 알림 채널에 동시 전송"""
        for notifier in self.notifiers:
            try:
                notifier.notify(self.order.customer, message)
            except Exception as e:
                # 하나의 알림 실패가 전체 흐름에 영향 없음
                logger.error(f"Notification failed: {str(e)}")

## 핵심 개념 정리

### 🎯 SRP (Single Responsibility Principle)
- **정의**: 하나의 클래스는 **하나의 변경 이유**만 가져야 함
- **현재 문제**: OrderService는 5가지 변경 이유 보유
  - 결제 방식 변경
  - 알림 채널 추가
  - 배달 시스템 변경
  - 저장소 변경
  - 주문 규칙 변경
- **해결**: 각 책임을 **별도 인터페이스**로 분리

### 🔓 OCP (Open-Closed Principle)
- **정의**: 확장에는 열려있고, 수정에는 닫혀있어야 함
