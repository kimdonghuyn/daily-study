## 모범 답안

### 1단계: 상속으로 중복 코드 제거

```python
# 부모 클래스: 모든 가게의 공통 기능 정의
class Store:
    def __init__(self, name, location):
        self.name = name
        self.location = location
        self.orders = []
    
    # 모든 가게가 동일하게 실행되는 메서드
    def accept_order(self, order):
        self.orders.append(order)
        return f"{self.name}에서 주문 {order.id}를 접수했습니다."
    
    def start_delivery(self, order):
        order.status = "배달중"
        return f"주문 {order.id}의 배달을 시작합니다."
    
    def complete_order(self, order):
        order.status = "완료"
        return f"주문 {order.id}가 완료되었습니다."
    
    # 각 가게별로 다른 구현이 필요한 메서드 (다형성)
    def calculate_delivery_fee(self, distance):
        raise NotImplementedError("각 가게 타입별로 구현 필요")
    
    def get_business_hours(self):
        raise NotImplementedError("각 가게 타입별로 구현 필요")
    
    def apply_cancellation_policy(self, order):
        raise NotImplementedError("각 가게 타입별로 구현 필요")
```

### 2단계: 다형성으로 각 가게별 특화 기능 구현

```python
# 자식 클래스들: 각 가게의 고유 특성 정의

class Restaurant(Store):
    def calculate_delivery_fee(self, distance):
        # 음식점: 기본 2,500원 + km당 300원
        return 2500 + (distance * 300)
    
    def get_business_hours(self):
        return "10:00 ~ 22:00"
    
    def apply_cancellation_policy(self, order):
        # 음식점: 준비 시작 후 취소 불가
        if order.status == "준비중":
            raise Exception("준비 중인 주문은 취소할 수 없습니다.")
        return True

class Cafe(Store):
    def calculate_delivery_fee(self, distance):
        # 카페: 기본 1,500원 + km당 200원 (음식점보다 저렴)
        return 1500 + (distance * 200)
    
    def get_business_hours(self):
        return "07:00 ~ 20:00"
    
    def apply_cancellation_policy(self, order):
        # 카페: 음료 준비가 빠르므로 1분 내 취소만 가능
        if order.created_time < datetime.now() - timedelta(minutes=1):
            raise Exception("1분 이내만 취소 가능합니다.")
        return True

class Convenience(Store):
    def calculate_delivery_fee(self, distance):
        # 편의점: 3km까지 무료, 이후 km당 500원
        if distance <= 3:
            return 0
        return (distance - 3) * 500
    
    def get_business_hours(self):
        return "24시간"
    
    def apply_cancellation_policy(self, order):
        # 편의점: 시간 제한 없이 취소 가능
        return True

class Pharmacy(Store):
    def calculate_delivery_fee(self, distance):
        # 약국: 거리 무관 고정 1,000원
        return 1000
    
    def get_business_hours(self):
        return "09:00 ~ 18:00"
    
    def apply_cancellation_policy(self, order):
        # 약국: 처방약은 취소 불가
        if order.has_prescription:
            raise Exception("처방약 주문은 취소할 수 없습니다.")
        return True
```

### 3단계: 다형성으로 통일된 인터페이스 사용

```python
# 핵심! 모든 가게를 동일하게 처리
class DeliveryService:
    def process_orders(self, stores: list[Store]):
        """
        가게 타입에 상관없이 동일한 로직으로 처리
        -> 새 가게 타입 추가 시 이 코드 수정 불필요!
        """
        for store in stores:
            print(f"=== {store.name} ===")
            
            # 같은 메서드 호출, 하지만 각 가게별로 다른 구현 실행
            print(f"영업시간: {store.get_business_hours()}")
            
            # 10km 배달비 조회 (같은 메서드, 다른 결과)
            fee = store.calculate_delivery_fee(distance=10)
            print(f"배달비: {fee:,}원")
            print()

# 사용 예시
service = DeliveryService()

# 4개의 다른 가게 타입을 동일한 로직으로 처리
stores = [
    Restaurant("맛있는 음식점", "강남"),
    Cafe("아늑한 카페", "강남"),
    Convenience("편의점 24", "강남"),
    Pharmacy("약국", "강남")
]

service.process_orders(stores)

# 출력:
# === 맛있는 음식점 ===
# 영업시간: 10:00 ~ 22:00
# 배달비: 5,500원
#
# === 아늑한 카페 ===
# 영업시간: 07:00 ~ 20:00
# 배달비: 3,500원
#
# === 편의점 24 ===
# 영업시간: 24시간
# 배달비: 0원
#
# === 약국 ===
# 영업시간: 09:00 ~ 18:00
# 배달비: 1,000원
```

### 4단계: 새 가게 타입 추가 (확장성)

```python
# 새로운 가게 타입 추가 필요? -> Store만 상속하면 됨!
# DeliveryService 코드는 수정 불필요! (Open-Closed Principle)

class Supermarket(Store):
    def calculate_delivery_fee(self, distance):
        return 3000 + (distance * 400)
    
    def get_business_hours(self):
        return "08:00 ~ 23:00"
    
    def apply_cancellation_policy(self, order):
        if order.status in ["준비중", "배달중"]:
            raise Exception("이 상태에서는 취소 불가")
        return True

# 바로 기존 코드에서 사용 가능!
stores.append(Supermarket("홈플러스", "강남"))
service.process_orders(stores)  # 수정 필요 없음!
```

## 핵심 개념 정리

### **상속 (Inheritance)**
- **목적**: 공통 기능을 부모 클래스에 정의하여 **중복 코드 제거**
- **구조**: 부모 클래스(Store) → 자식 클래스들(Restaurant, Cafe...)
- **이점**: 
  - 코드 재사용으로 개발 속도 향상
  - 공통 로직 수정 시 한 곳만 변경 (유지보수성 ↑)

### **다형성 (Polymorphism)**
- **목적**: **같은 메서드 호출**이지만 각 클래스별로 **다른 구현 실행**
- **메커니즘**: 
  - `calculate_delivery_fee()` 같은 메서드를 각 가게별로 오버라이드
  - 런타임에 실제 객체 타입에 맞는 메서드 실행 (동적 바인딩)
- **이점**:
  - 클라이언트 코드(DeliveryService)는 Store 타입만 알면 됨
  - 새 가게 타입 추가 시 기존 코드 수정 불필요

### **메서드 오버라이딩 (Method Overriding)**
```python
# 부모: 구현 강제 (abstract method)
class Store:
    def calculate_delivery_fee(self, distance):
        raise NotImplementedError()

# 자식: 각자 구현
class Restaurant(Store):
    def calculate_delivery_fee(self, distance):
        return 2500 + (distance * 300)  # 음식점식 계산
```
