### 1. 🔧 문제 상황

# 상속과 다형성을 활용한 결제 시스템 개선

## 모범 답안

### 문제의 본질
현재 코드는 **새로운 결제 수단이 추가될 때마다 기존 코드를 수정해야 하는 구조(Open-Closed Principle 위반)**입니다. 이는 버그 유입 위험과 유지보수 비용을 증가시킵니다.

### 상속과 다형성을 활용한 해결책

```python
from abc import ABC, abstractmethod
from typing import Dict

# 1. 추상 기본 클래스 정의 (인터페이스 역할)
class PaymentMethod(ABC):
    """모든 결제 수단이 구현해야 할 계약"""
    
    @abstractmethod
    def validate(self) -> bool:
        """결제 수단 유효성 검사"""
        pass
    
    @abstractmethod
    def charge(self, amount: int) -> bool:
        """실제 결제 처리"""
        pass
    
    @abstractmethod
    def refund(self, transaction_id: str, amount: int) -> bool:
        """환불 처리"""
        pass

# 2. 구체적인 결제 수단 구현 (상속)
class CreditCardPayment(PaymentMethod):
    def __init__(self, card_number: str, cvv: str):
        self.card_number = card_number
        self.cvv = cvv
    
    def validate(self) -> bool:
        # 신용카드 유효성 검사 로직
        return len(self.card_number) == 16 and len(self.cvv) == 3
    
    def charge(self, amount: int) -> bool:
        if not self.validate():
            raise ValueError("Invalid card")
        # 신용카드 결제 API 호출
        print(f"신용카드로 {amount}원 결제 중...")
        return True
    
    def refund(self, transaction_id: str, amount: int) -> bool:
        print(f"신용카드 {amount}원 환불 처리...")
        return True

class MobilePayment(PaymentMethod):
    def __init__(self, phone_number: str):
        self.phone_number = phone_number
    
    def validate(self) -> bool:
        return len(self.phone_number) == 11
    
    def charge(self, amount: int) -> bool:
        if not self.validate():
            raise ValueError("Invalid phone number")
        print(f"휴대폰으로 {amount}원 결제 중...")
        return True
    
    def refund(self, transaction_id: str, amount: int) -> bool:
        print(f"휴대폰 {amount}원 환불 처리...")
        return True

class KakaoPay(PaymentMethod):
    def __init__(self, user_id: str):
        self.user_id = user_id
    
    def validate(self) -> bool:
        return bool(self.user_id)
    
    def charge(self, amount: int) -> bool:
        if not self.validate():
            raise ValueError("Invalid user")
        print(f"카카오페이로 {amount}원 결제 중...")
        return True
    
    def refund(self, transaction_id: str, amount: int) -> bool:
        print(f"카카오페이 {amount}원 환불 처리...")
        return True

# 새로운 결제 수단 추가 (기존 코드 수정 불필요!)
class NaverPay(PaymentMethod):
    def __init__(self, naver_id: str):
        self.naver_id = naver_id
    
    def validate(self) -> bool:
        return bool(self.naver_id)
    
    def charge(self, amount: int) -> bool:
        if not self.validate():
            raise ValueError("Invalid Naver ID")
        print(f"네이버페이로 {amount}원 결제 중...")
        return True
    
    def refund(self, transaction_id: str, amount: int) -> bool:
        print(f"네이버페이 {amount}원 환불 처리...")
        return True

# 3. 결제 시스템 (다형성 활용)
class PaymentProcessor:
    """모든 결제 수단을 동일하게 처리"""
    
    def __init__(self):
        self.transaction_history: Dict[str, dict] = {}
    
    def process_payment(
        self, 
        payment_method: PaymentMethod, 
        amount: int, 
        order_id: str
    ) -> bool:
        """다형성: PaymentMethod 타입의 어떤 객체든 처리 가능"""
        try:
            if payment_method.charge(amount):
                self.transaction_history[order_id] = {
                    'amount': amount,
                    'method': payment_method.__class__.__name__,
                    'status': 'completed'
                }
                return True
        except Exception as e:
            print(f"결제 실패: {e}")
            return False
    
    def process_refund(
        self, 
        payment_method: PaymentMethod, 
        transaction_id: str, 
        amount: int
    ) -> bool:
        return payment_method.refund(transaction_id, amount)

# 4. 사용 예시
if __name__ == "__main__":
    processor = PaymentProcessor()
    
    # 신용카드 결제
    card = CreditCardPayment("1234567890123456", "123")
    processor.process_payment(card, 50000, "order_001")
    
    # 휴대폰 결제
    mobile = MobilePayment("01012345678")
    processor.process_payment(mobile, 30000, "order_002")
    
    # 카카오페이
    kakao = KakaoPay("user123")
    processor.process_payment(kakao, 20000, "order_003")
    
    # 네이버페이 (새로 추가되어도 PaymentProcessor 수정 불필요!)
    naver = NaverPay("naver_id_456")
    processor.process_payment(naver, 15000, "order_004")
```

## 핵심 개념 정리

### 상속 (Inheritance)
- **정의**: 부모 클래스의 속성과 메서드를 자식 클래스가 물려받는 것
- **목적**: 공통 인터페이스 정의로 다형성의 기반 마련
- **실무 예**: `PaymentMethod` → `CreditCardPayment`, `MobilePayment` 등

### 다형성 (Polymorphism)
- **정의**: 같은 메서드 호출이 객체에 따라 다르게 작동하는 성질
- **종류**:
  - **컴파일 타임**: 메서드 오버로딩 (Python은 미지원)
  - **런타임**: 메서드 오버라이딩 (이 문제의 핵심)
- **이점**: 
  - 새로운 타입 추가 시 기존 코드 수정 불필요
  - 결합도(coupling) 감소
  - 코드 확장성 극대화

### Open-Closed Principle (OCP)
- **정의**: "확장에는 열려있고, 수정에는 닫혀있어야 한다"
- **위반 사례**: 새 결제 수단 추가 시 기존 if문 수정 필요
- **준수 방법**: 추상 클래스 + 다형성으로 확장 가능하게 설계

### 리스코프 치환 원칙 (LSP)
- **정의**: 부모 클래스 타입으로 자식 클래스 사용 가능
- **예시**: `PaymentProcessor.process_payment(payment_method: PaymentMethod)`
  - 어떤 PaymentMethod 구현체든 대체 가능
