## 📚 모범 답안 & 개념 정리

### 1. 🔧 문제 상황

# 상속과 다형성을 활용한 결제 시스템 설계

## 모범 답안

**상속과 다형성은 이 문제를 "개방-폐쇄 원칙(Open-Closed Principle)"으로 해결합니다.**

### 핵심 아이디어
1. **추상 기본 클래스(또는 인터페이스)로 공통 계약 정의**
   - 모든 결제 수단이 따라야 할 `pay()`, `validate()`, `approve()` 등의 메서드 정의
   
2. **각 결제 수단별로 구체적인 클래스 상속**
   - 공통 로직은 기본 클래스에서 처리
   - 특화된 로직(카드번호 검증, 토큰 확인 등)은 각 클래스에서 override

3. **다형성으로 결제 수단을 통일된 인터페이스로 처리**
   - if-else 제거
   - 새로운 결제 수단 추가 시 기존 코드 수정 불필요

### 구현 예시

```python
# 1. 추상 기본 클래스 - 결제 수단의 계약 정의
from abc import ABC, abstractmethod
from datetime import datetime

class PaymentMethod(ABC):
    """모든 결제 수단이 따라야 할 기본 계약"""
    
    def __init__(self, amount: float):
        self.amount = amount
        self.transaction_id = None
        self.status = "PENDING"
    
    @abstractmethod
    def validate(self) -> bool:
        """각 결제 수단별로 유효성 검증"""
        pass
    
    @abstractmethod
    def process_payment(self) -> dict:
        """실제 결제 처리"""
        pass
    
    def approve(self) -> bool:
        """공통 로직 - 모든 결제 수단이 동일"""
        print(f"결제 승인 대기 중... ({self.amount}원)")
        self.status = "APPROVED"
        return True
    
    def execute(self) -> dict:
        """결제 실행 전체 흐름 - Template Method Pattern"""
        if not self.validate():
            self.status = "FAILED"
            return {"success": False, "error": "유효성 검증 실패"}
        
        if not self.approve():
            self.status = "FAILED"
            return {"success": False, "error": "승인 실패"}
        
        return self.process_payment()


# 2. 신용카드 결제
class CreditCardPayment(PaymentMethod):
    def __init__(self, amount: float, card_number: str, cvv: str):
        super().__init__(amount)
        self.card_number = card_number
        self.cvv = cvv
    
    def validate(self) -> bool:
        """신용카드 특화: 카드번호와 CVV 검증"""
        # 카드번호 길이 확인 (간단한 예)
        if len(self.card_number) != 16:
            print("❌ 유효하지 않은 카드번호")
            return False
        
        if len(self.cvv) != 3:
            print("❌ 유효하지 않은 CVV")
            return False
        
        print("✅ 신용카드 유효성 검증 완료")
        return True
    
    def process_payment(self) -> dict:
        """신용카드 결제 처리"""
        self.transaction_id = f"CC-{datetime.now().strftime('%Y%m%d%H%M%S')}"
        print(f"💳 신용카드로 {self.amount}원 결제 처리 중...")
        self.status = "COMPLETED"
        return {
            "success": True,
            "method": "Credit Card",
            "amount": self.amount,
            "transaction_id": self.transaction_id
        }


# 3. 카카오페이 결제
class KakaoPayPayment(PaymentMethod):
    def __init__(self, amount: float, token: str):
        super().__init__(amount)
        self.token = token
    
    def validate(self) -> bool:
        """카카오페이 특화: 토큰 유효성 검증"""
        # 토큰이 실제로 유효한지 카카오 서버에 확인 (예시)
        if not self.token or len(self.token) < 10:
            print("❌ 유효하지 않은 카카오페이 토큰")
            return False
        
        print("✅ 카카오페이 토큰 검증 완료")
        return True
    
    def process_payment(self) -> dict:
        """카카오페이 결제 처리"""
        self.transaction_id = f"KP-{datetime.now().strftime('%Y%m%d%H%M%S')}"
        print(f"📱 카카오페이로 {self.amount}원 결제 처리 중...")
        self.status = "COMPLETED"
        return {
            "success": True,
            "method": "Kakao Pay",
            "amount": self.amount,
            "transaction_id": self.transaction_id
        }


# 4. 계좌이체 결제
class BankTransferPayment(PaymentMethod):
    def __init__(self, amount: float, bank_code: str, account: str):
        super().__init__(amount)
        self.bank_code = bank_code
        self.account = account
    
    def validate(self) -> bool:
        """계좌이체 특화: 계좌 유효성 검증"""
        if not self.account or len(self.account) < 10:
            print("❌ 유효하지 않은 계좌번호")
            return False
        
        print("✅ 계좌이체 정보 검증 완료")
        return True
    
    def process_payment(self) -> dict:
        """계좌이체 결제 처리"""
        self.transaction_id = f"BT-{datetime.now().strftime('%Y%m%d%H%M%S')}"
        print(f"🏦 계좌이체로 {self.amount}원 결제 처리 중...")
        self.status = "COMPLETED"
        return {
            "success": True,
            "method": "Bank Transfer",
            "amount": self.amount,
            "transaction_id": self.transaction_id
        }


# 5. 결제 시스템 - 다형성의 진정한 가치
class PaymentProcessor:
    """어떤 결제 수단이든 통일된 인터페이스로 처리"""
    
    def __init__(self):
        self.payment_history = []
    
    def process(self, payment: PaymentMethod) -> dict:
        """
        💡 핵심: payment의 실제 타입(신용카드, 카카오페이 등)을 
        신경 쓰지 않고 PaymentMethod 인터페이스로만 처리
        """
        result = payment.execute()
        self.payment_history.append(result)
        return result
    
    def show_history(self):
        """결제 히스토리 조회"""
        for record in self.payment_history:
            print(f"- {record['method']}: {record['amount']}원 "
                  f"(ID: {record['transaction_id']})")


# ============ 실제 사용 ============

def main():
    processor = PaymentProcessor()
    
    print("=== 신용카드 결제 ===")
    card_payment = CreditCardPayment(
        amount=50000,
        card_number="1234567890123456",
        cvv="123"
    )
    processor.process(card_payment)
    
    print("\n=== 카카오페이 결제 ===")
    kakao_payment = KakaoPayPayment(
        amount=30000,
        token="kakao_token_1234567890"
    )
    processor.process(kakao_payment)
    
    print("\n=== 계좌이체 결제 ===")
    transfer_payment = BankTransferPayment(
        amount=100000,
        bank_code="004",
        account="12345678901234"
    )
    processor.process(transfer_payment)
    
    print("\n=== 결제 히스토리 ===")
    processor.show_history()


if __name__ == "__main__":
    main()
```
