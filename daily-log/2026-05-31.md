### 메인 질문: SRP를 통한 역할 분리

**현재 상태의 문제점:**
바리스타가 모든 책임을 지고 있어서:
- 주문 실수 → 커피 제조 지연
- 계산 실수 → 매출 오류
- 한 업무에서의 변경이 다른 업무에 영향

**SRP 적용 - 역할 분리:**

```python
# ❌ SRP 위반: 하나의 클래스가 여러 책임을 짐
class Barista:
    def take_order(self, customer, item):
        """주문 받기"""
        pass
    
    def make_coffee(self, item):
        """커피 만들기"""
        pass
    
    def process_payment(self, amount):
        """결제 처리"""
        pass
    
    def print_receipt(self, order):
        """영수증 출력"""
        pass

# ✅ SRP 적용: 책임을 분리
class OrderTaker:
    """주문 담당"""
    def take_order(self, customer, item):
        return Order(customer, item)

class Coffeemaker:
    """커피 제조 담당"""
    def make_coffee(self, order):
        print(f"{order.item} 제조 시작")
        return Coffee(order.item)

class Cashier:
    """결제/영수증 담당"""
    def process_payment(self, order, amount):
        if self.validate_payment(amount, order.price):
            return Receipt(order, amount)
    
    def print_receipt(self, receipt):
        print(receipt)

class CafeManager:
    """조율 역할"""
    def __init__(self):
        self.order_taker = OrderTaker()
        self.coffee_maker = Coffeemaker()
        self.cashier = Cashier()
    
    def serve_customer(self, customer, item):
        order = self.order_taker.take_order(customer, item)
        coffee = self.coffee_maker.make_coffee(order)
        receipt = self.cashier.process_payment(order, customer.payment)
        self.cashier.print_receipt(receipt)
        return coffee
```

**SRP의 이점:**
- 각 클래스는 변경의 이유가 1개만 존재
- 주문 로직 변경 → OrderTaker만 수정
- 결제 로직 변경 → Cashier만 수정
- 테스트와 유지보수가 용이

```python
# ✅ OCP 적용: 추상화를 통한 확장
from abc import ABC, abstractmethod

# 결제 방식의 추상 인터페이스
class PaymentMethod(ABC):
    @abstractmethod
    def pay(self, amount: int) -> bool:
        """결제 처리"""
        pass
    
    @abstractmethod
    def refund(self, amount: int) -> bool:
        """환불 처리"""
        pass

# 구체적인 결제 수단들
class CashPayment(PaymentMethod):
    def pay(self, amount):
        print(f"현금 {amount}원 결제")
        return True
    
    def refund(self, amount):
        print(f"현금 {amount}원 환불")
        return True

class CardPayment(PaymentMethod):
    def __init__(self, card_number):
        self.card_number = card_number
    
    def pay(self, amount):
        print(f"카드 {self.card_number} 로 {amount}원 결제")
        return self.validate_card()
    
    def refund(self, amount):
        print(f"카드 환불 {amount}원")
        return True
    
    def validate_card(self):
        return len(self.card_number) == 16

class KakaoPayPayment(PaymentMethod):
    def __init__(self, user_id):
        self.user_id = user_id
    
    def pay(self, amount):
        print(f"카카오페이 사용자 {self.user_id} 결제 {amount}원")
        return self.check_balance(amount)
    
    def refund(self, amount):
        print(f"카카오페이 {amount}원 환불")
        return True
    
    def check_balance(self, amount):
        # 실제로는 카카오페이 API 호출
        return True

class NaverPayPayment(PaymentMethod):
    def __init__(self, user_id):
        self.user_id = user_id
    
    def pay(self, amount):
        print(f"네이버페이 사용자 {self.user_id} 결제 {amount}원")
        return True
    
    def refund(self, amount):
        print(f"네이버페이 {amount}원 환불")
        return True

# 새로운 결제 수단 추가는 기존 코드 수정 없이 확장
class ApplePayPayment(PaymentMethod):
    def __init__(self, apple_id):
        self.apple_id = apple_id
    
    def pay(self, amount):
        print(f"애플페이 {self.apple_id} 결제 {amount}원")
        return True
    
    def refund(self, amount):
        print(f"애플페이 {amount}원 환불")
        return True

# 수정되지 않는 Cashier 클래스
class Cashier:
    """기존 코드는 전혀 수정되지 않음!"""
    def process_payment(self, order: 'Order', payment_method: PaymentMethod) -> 'Receipt':
        if payment_method.pay(order.total_price):
            return Receipt(order, payment_method)
        else:
            raise PaymentException("결제 실패")
    
    def refund_order(self, receipt: 'Receipt', amount: int):
        if receipt.payment_method.refund(amount):
            print(f"환불 성공: {amount}원")

# 사용 예시
def main():
    order = Order("아메리카노", 4500)
    cashier = Cashier()
    
    # 기존 결제 수단
    cashier.process_payment(order, CashPayment())
    cashier.process_payment(order, CardPayment("1234567890123456"))
    cashier.process_payment(order, KakaoPayPayment("user123"))
    
    # 새로운 결제 수단 추가 (기존 코드 수정 없음!)
    cashier.process_payment(order, ApplePayPayment("apple@example.com"))
```

**OCP의 이점:**
- 새로운 결제 수단 추가 시 `PaymentMethod` 상속하면 끝
- Cashier 클래스는 건드리지 않음 (수정에 닫혀있음)
- 새로운 결제 수단 추가 가능 (확장에 열려있음)

## 핵심 개념 정리

### SRP (Single Responsibility Principle)

- **정의**: 클래스는 하나의 책임만 가져야 함
- **변경의 이유**: 클래스가 변경되는 이유는 단 하나여야 함
- **이점**:
  - 코드 응집도 ⬆️ (관련 기능이 모여있음)
  - 클래스 간 결합도 ⬇️
  - 단위 테스트 용이
  - 재사용성 증가
- **체크리스트**:
  - 이 클래스를 왜 변경해야 하나? (2개 이상이면 SRP 위반)
  - 이 클래스의 책임을 한 문장으로 설명할 수 있나?
