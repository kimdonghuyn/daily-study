### 1. 🔧 문제 상황

# 음식 배달 앱 - 상속과 다형성 설계

## 모범 답안

이 상황은 **상속과 다형성**을로 해결하는 전형적인 사례입니다.

### 1️⃣ 문제 분석
- 모든 음식이 **동일한 프로세스**를 따름 (주문 → 조리 → 배송)
- 음식별로 **구체적인 구현만 다름** (조리 시간, 배송 방식 등)
- 새로운 음식 추가 시마다 코드 중복 발생

### 2️⃣ 해결 방안: 상속 기반 설계

```python
# 1. 부모 클래스: 공통 프로세스 정의
from abc import ABC, abstractmethod
from typing import List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Order:
    order_id: str
    customer_name: str
    address: str
    quantity: int
    status: str = "주문 접수"

class Food(ABC):
    """모든 음식이 따를 공통 인터페이스"""
    
    def __init__(self, name: str, base_price: int):
        self.name = name
        self.base_price = base_price
    
    # 공통 플로우 (Template Method Pattern)
    def process_order(self, order: Order) -> dict:
        """주문 처리 공통 절차"""
        print(f"\n📋 주문 처리 시작: {self.name}")
        
        # 1단계: 주문 수락
        self.accept_order(order)
        
        # 2단계: 조리
        cooking_time = self.cook(order)
        
        # 3단계: 품질 검사
        self.quality_check(order)
        
        # 4단계: 배송
        self.deliver(order)
        
        return {
            'order_id': order.order_id,
            'food': self.name,
            'total_price': self.calculate_price(order),
            'cooking_time': cooking_time,
            'status': '배송 완료'
        }
    
    # 공통 메서드
    def accept_order(self, order: Order):
        order.status = "주문 수락"
        print(f"✅ {order.order_id} 주문 수락됨")
    
    # 각 음식별로 구현해야 할 메서드
    @abstractmethod
    def cook(self, order: Order) -> int:
        """조리하기 - 각 음식별로 다른 로직"""
        pass
    
    @abstractmethod
    def quality_check(self, order: Order):
        """품질 검사 - 각 음식별로 다른 기준"""
        pass
    
    @abstractmethod
    def deliver(self, order: Order):
        """배송하기 - 각 음식별로 다른 방식"""
        pass
    
    def calculate_price(self, order: Order) -> int:
        """가격 계산 - 기본 구현 제공"""
        return self.base_price * order.quantity


# 2. 자식 클래스: 각 음식별 구체적 구현
class Pizza(Food):
    def cook(self, order: Order) -> int:
        cooking_time = 25  # 분
        print(f"🍕 피자 조리 중... ({cooking_time}분 소요)")
        order.status = "조리 중"
        return cooking_time
    
    def quality_check(self, order: Order):
        print("✨ 피자 온도 확인: 180°C 유지 ✓")
        order.status = "품질 검사 완료"
    
    def deliver(self, order: Order):
        print(f"🚗 피자 배송 중 → {order.address}")
        order.status = "배송 중"


class Chicken(Food):
    def cook(self, order: Order) -> int:
        cooking_time = 20
        print(f"🍗 치킨 튀김 중... ({cooking_time}분 소요)")
        order.status = "조리 중"
        return cooking_time
    
    def quality_check(self, order: Order):
        print("✨ 치킨 바삭함 확인 ✓")
        order.status = "품질 검사 완료"
    
    def deliver(self, order: Order):
        print(f"🏍️ 오토바이 배송 → {order.address}")
        order.status = "배송 중"
    
    # 치킨만의 추가 기능
    def calculate_price(self, order: Order) -> int:
        base_price = super().calculate_price(order)
        sauce_fee = 2000  # 소스 추가료
        return base_price + sauce_fee


class Burger(Food):
    def cook(self, order: Order) -> int:
        cooking_time = 15
        print(f"🍔 버거 조리 중... ({cooking_time}분 소요)")
        order.status = "조리 중"
        return cooking_time
    
    def quality_check(self, order: Order):
        print("✨ 버거 신선도 확인 ✓")
        order.status = "품질 검사 완료"
    
    def deliver(self, order: Order):
        print(f"🚲 자전거 배송 → {order.address}")
        order.status = "배송 중"


# 3. 주문 처리 시스템 (다형성 활용)
class OrderProcessingSystem:
    def __init__(self):
        # 음식 카탈로그
        self.foods: dict = {
            'pizza': Pizza('피자', 18000),
            'chicken': Chicken('치킨', 15000),
            'burger': Burger('버거', 12000),
        }
    
    def add_new_food(self, food_type: str, food: Food):
        """새로운 음식 추가 - 기존 코드 수정 불필요!"""
        self.foods[food_type] = food
    
    def process_order(self, food_type: str, order: Order) -> dict:
        """어떤 음식이든 동일한 처리 방식"""
        if food_type not in self.foods:
            raise ValueError(f"지원하지 않는 음식: {food_type}")
        
        food = self.foods[food_type]
        return food.process_order(order)  # 다형성: 실제 음식 타입에 따라 다른 메서드 실행
    
    def process_batch_orders(self, orders: List[tuple]) -> List[dict]:
        """대량 주문 처리"""
        results = []
        for food_type, order in orders:
            result = self.process_order(food_type, order)
            results.append(result)
        return results


# 4. 사용 예시
if __name__ == "__main__":
    system = OrderProcessingSystem()
    
    # 기본 주문 처리
    pizza_order = Order("ORD001", "김철수", "서울 강남구", 2)
    result = system.process_order('pizza', pizza_order)
    print(f"\n결과: {result}")
    
    print("\n" + "="*50)
    
    # 새로운 음식 추가 (기존 코드 수정 없음!)
    class Chinese(Food):
        def cook(self, order: Order) -> int:
            cooking_time = 30
            print(f"🥢 중식 조리 중... ({cooking_time}분 소요)")
            return cooking_time
        
        def quality_check(self, order: Order):
            print("✨ 중식 맛 확인 ✓")
        
        def deliver(self, order: Order):
            print(f"🛵 배송 → {order.address}")
    
    system.add_new_food('chinese', Chinese('중식', 14000))
    
    chinese_order = Order("ORD002", "이영희", "서울 마포구", 1)
    result = system.process_order('chinese', chinese_order)
    print(f"\n결과: {result}")
    
    # 대량 주문 처리
    print("\n" + "="*50)
    print("🎯 대량 주문 처리")
    batch = [
        ('pizza', Order("ORD003", "박민준", "서울 서초구", 1)),
        ('chicken', Order("ORD004", "최수진", "서울 종로구", 3)),
        ('burger', Order("ORD005", "정준호", "서울 강동구", 2)),
    ]
    results = system.process_batch_orders(batch)
    
    for result in results:
        print(f"주문 {result['order_id']}: {result['food']} - {result['total_price']:,}원")
```
