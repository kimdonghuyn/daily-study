### 1. 🔧 문제 상황

# 상속과 다형성 - 게임 캐릭터 설계 문제

## 모범 답안

**상속(Inheritance)과 다형성(Polymorphism)을 활용한 구조화**

```python
# 1단계: 공통 부분을 추상 클래스(Base Class)로 정의
from abc import ABC, abstractmethod

class Character(ABC):
    """모든 캐릭터의 기본 구조를 정의"""
    
    def __init__(self, name: str, hp: int, speed: int):
        self.name = name
        self.hp = hp
        self.max_hp = hp
        self.speed = speed
    
    # ✅ 공통 메서드: 모든 캐릭터에서 동일
    def take_damage(self, damage: int):
        """체력 감소 로직 - 한 곳에서만 관리"""
        self.hp = max(0, self.hp - damage)
        print(f"{self.name}이(가) {damage}의 피해를 입었습니다. (남은 체력: {self.hp})")
    
    def move(self, distance: int):
        """이동 - 기본 구현"""
        print(f"{self.name}이(가) {distance}칸 이동했습니다.")
    
    # ❌ 추상 메서드: 각 캐릭터마다 다르게 구현 필수
    @abstractmethod
    def attack(self, target: 'Character'):
        """공격 방식은 캐릭터마다 다름"""
        pass
    
    @abstractmethod
    def get_attack_description(self) -> str:
        """공격 설명"""
        pass


# 2단계: 각 캐릭터가 공통 클래스를 상속받아 고유 부분만 구현
class Player(Character):
    def __init__(self, name: str):
        super().__init__(name, hp=100, speed=5)
        self.weapon = "검"
    
    def attack(self, target: Character):
        damage = 25
        print(f"{self.name}이(가) {self.weapon}으로 공격합니다!")
        target.take_damage(damage)
    
    def get_attack_description(self) -> str:
        return f"{self.name}의 {self.weapon} 공격"


class Monster(Character):
    def __init__(self, name: str):
        super().__init__(name, hp=30, speed=3)
        self.attack_power = 10
    
    def attack(self, target: Character):
        damage = self.attack_power
        print(f"{self.name}이(가) 발톱으로 할퀴며 공격합니다!")
        target.take_damage(damage)
    
    def get_attack_description(self) -> str:
        return f"{self.name}의 발톱 공격"


class Boss(Character):
    def __init__(self, name: str):
        super().__init__(name, hp=200, speed=2)
        self.mana = 100
    
    def attack(self, target: Character):
        damage = 50
        print(f"{self.name}이(가) 마법을 시전합니다! ✨")
        target.take_damage(damage)
        self.mana -= 20
    
    def get_attack_description(self) -> str:
        return f"{self.name}의 마법 공격"


class NPC(Character):
    def __init__(self, name: str):
        super().__init__(name, hp=50, speed=2)
    
    def attack(self, target: Character):
        # NPC는 약하게 공격
        damage = 5
        print(f"{self.name}이(가) 약하게 공격합니다.")
        target.take_damage(damage)
    
    def get_attack_description(self) -> str:
        return f"{self.name}의 약한 공격"


# 3단계: 다형성 활용 - 같은 인터페이스로 다양한 캐릭터 제어
def battle_sequence(attacker: Character, defender: Character):
    """캐릭터 타입과 관계없이 동일한 로직으로 전투 처리"""
    print(f"\n>>> {attacker.get_attack_description()}")
    attacker.attack(defender)
    print(f">>> {defender.name}의 상태: HP {defender.hp}/{defender.max_hp}")


# 실행 예제
if __name__ == "__main__":
    # 각 캐릭터 생성
    player = Player("용사")
    monster = Monster("고블린")
    boss = Boss("드래곤")
    npc = NPC("마을 주민")
    
    characters: list[Character] = [player, monster, boss, npc]
    
    # ✅ 같은 코드로 모든 캐릭터 처리
    print("=== 전투 시작 ===")
    battle_sequence(player, monster)
    battle_sequence(monster, player)
    battle_sequence(boss, player)
    battle_sequence(player, boss)
    
    # ✅ 체력 규칙 변경시 Character 클래스의 take_damage() 메서드만 수정
    print("\n=== 모든 캐릭터의 현재 상태 ===")
    for char in characters:
        print(f"{char.name}: HP {char.hp}/{char.max_hp}")
```

## 핵심 개념 정리

### 📌 **상속(Inheritance)**
- **정의**: 부모 클래스의 속성과 메서드를 자식 클래스가 물려받음
- **목적**: 코드 재사용 및 공통 부분 중앙화
- **구현**: `class Player(Character)` - Character의 모든 것을 상속
- **장점**: 중복 제거, 유지보수 용이

### 📌 **다형성(Polymorphism)**
- **정의**: 같은 메서드 이름이 객체마다 다르게 동작
- **메커니즘**: `attack()` 메서드는 공통이지만 각 클래스에서 다르게 구현
- **실제 효과**: `attacker.attack(target)`은 attacker의 실제 타입에 따라 다르게 동작
- **이점**: 새로운 캐릭터 타입 추가 시 전투 로직 수정 불필요

### 📌 **추상 클래스(Abstract Class)**
```python
from abc import ABC, abstractmethod

class Character(ABC):
    @abstractmethod
    def attack(self, target):
        pass  # 반드시 구현해야 함
```
- 일부 메서드는 "반드시 구현해야 하는 틀"로만 정의
- 자식 클래스가 해당 메서드를 구현하도록 강제

### 📌 **메서드 오버라이딩(Method Overriding)**
- 부모의 메서드를 자식이 자신의 방식으로 재구현
- `super().__init__()` - 부모 생성자 호출
