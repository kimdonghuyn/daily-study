### 해결책 (상속 + 다형성)
```python
from abc import ABC, abstractmethod
from datetime import datetime
from typing import Dict

# 1️⃣ 기본 클래스 (상속의 기초)
class FoodOrder(ABC):
    """모든 음식 주문의 공통 구조를 정의"""
    
    def __init__(self, customer_id: str, address: str, base_price: float):
        self.customer_id = customer_id
        self.address = address
        self.base_price = base_price
        self.created_at = datetime.now()
    
    # 공통 로직 - 반복 제거
    def validate(self) -> None:
        """모든 주문이 공통으로 검증해야 할 사항"""
        if not self.customer_id:
            raise ValueError("고객 ID는 필수입니다")
        if not self.address:
            raise ValueError("배송지는 필수입니다")
        if self.base_price <= 0:
            raise ValueError("가격은 0보다 커야 합니다")
        self._validate_specific()  # 각 음식별 추가 검증
    
    def create_order(self) -> Dict:
        """주문 생성 프로세스 (Template Method Pattern)"""
        self.validate()
        price = self.calculate_price()
        estimated_delivery = self.estimate_delivery_time()
        
        return {
            "order_id": self.generate_order_id(),
            "food_type": self.get_food_type(),
            "customer_id": self.customer_id,
            "price": price,
            "estimated_delivery": estimated_delivery,
            "status": "pending",
            "created_at": self.created_at.isoformat()
        }
    
    # 2️⃣ 다형성: 각 서브클래스가 오버라이드
    @abstractmethod
    def calculate_price(self) -> float:
        """각 음식별로 가격 계산 방식이 다름"""
        pass
    
    @abstractmethod
    def estimate_delivery_time(self) -> int:
        """각 음식별로 조리 시간이 다름"""
        pass
    
    @abstractmethod
    def get_food_type(self) -> str:
        """음식 타입 반환"""
        pass
    
    def _validate_specific(self) -> None:
        """각 음식별 추가 검증 (선택)"""
        pass
    
    def generate_order_id(self) -> str:
        """공통 유틸리티"""
        import uuid
        return f"{self.get_food_type()}_{uuid.uuid4().hex[:8]}"


# 2️⃣ 구체적인 음식 클래스들 (서브클래스)
class PizzaOrder(FoodOrder):
    """피자 주문 - 고유 로직만 구현"""
    
    def calculate_price(self) -> float:
        # 피자는 10% 배달료
        return self.base_price * 1.10
    
    def estimate_delivery_time(self) -> int:
        # 피자는 25분
        return 25
    
    def get_food_type(self) -> str:
        return "pizza"
    
    def _validate_specific(self) -> None:
        # 피자 고유 검증
        if self.base_price < 10000:
            raise ValueError("피자 최소 주문 금액은 10,000원입니다")


class ChickenOrder(FoodOrder):
    """치킨 주문"""
    
    def calculate_price(self) -> float:
        # 치킨은 15% 배달료
        return self.base_price * 1.15
    
    def estimate_delivery_time(self) -> int:
        # 치킨은 30분
        return 30
    
    def get_food_type(self) -> str:
        return "chicken"


class KoreanFoodOrder(FoodOrder):
    """한식 주문"""
    
    def calculate_price(self) -> float:
        # 한식은 12% 배달료
        return self.base_price * 1.12
    
    def estimate_delivery_time(self) -> int:
        # 한식은 20분
        return 20
    
    def get_food_type(self) -> str:
        return "korean"


class JapaneseFoodOrder(FoodOrder):
    """일식 주문 - 새로운 음식 추가 (확장성!)"""
    
    def calculate_price(self) -> float:
        # 일식은 20% 배달료 (고가)
        return self.base_price * 1.20
    
    def estimate_delivery_time(self) -> int:
        # 일식은 35분
        return 35
    
    def get_food_type(self) -> str:
        return "japanese"


# 3️⃣ 다형성의 힘 - 같은 인터페이스로 다양한 동작
class OrderProcessor:
    """주문 처리기 - 모든 음식 타입을 동일하게 처리"""
    
    def process_orders(self, orders: list[FoodOrder]) -> list[Dict]:
        """여러 주문을 동일한 방식으로 처리"""
        results = []
        for order in orders:
            # 각 order의 실제 타입에 따라 다른 동작이 일어남
            # 이게 다형성!
            try:
                result = order.create_order()
                results.append(result)
            except ValueError as e:
                results.append({
                    "error": str(e),
                    "food_type": order.get_food_type()
                })
        return results


# 🎯 사용 예시
if __name__ == "__main__":
    # 다양한 주문 생성
    orders = [
        PizzaOrder("cust_001", "서울시 강남구", 25000),
        ChickenOrder("cust_002", "서울시 마포구", 30000),
        KoreanFoodOrder("cust_003", "서울시 종로구", 20000),
        JapaneseFoodOrder("cust_004", "서울시 강서구", 50000),  # 새로운 음식 추가!
    ]
    
    # 동일한 프로세스로 처리
    processor = OrderProcessor()
    results = processor.process_orders(orders)
    
    for result in results:
        print(result)
        print("---")
```
